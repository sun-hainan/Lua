## 1. 运行时类型检查

### 代码示例

```lua
-- type() 函数
print("type of 42:", type(42))              -- number
print("type of 'hello':", type('hello'))    -- string
print("type of nil:", type(nil))            -- nil
print("type of {}", type({}))              -- table

-- 多态函数
local function process(val)
    local t = type(val)
    if t == "number" then
        print("processing number:", val * 2)
    elseif t == "string" then
        print("processing string:", val .. val)
    elseif t == "table" then
        print("processing table:", next(val) or "empty")
    elseif t == "function" then
        print("processing function:", val())
    end
end

process(42)
process("hi")
process(function() return 100 end)
```

## 2. 动态代码执行

### 概念讲解

`loadstring(s)` / `load(s)`：编译字符串为函数

### 代码示例

```lua
-- 执行字符串
local chunk, err = loadstring("return 2 + 3")
if chunk then
    print("result:", chunk())  -- 5
end

-- 沙箱环境
local function sandbox()
    local env = {
        print = print,
        math = math,
        io = nil,  -- 禁止 io
    }
    setmetatable(env, {__index = _G})

    local chunk = loadstring("print(math.random(10))", nil, env)
    if chunk then chunk() end
end

sandbox()
```

## 3. eval 实现

### 代码示例

```lua
local function eval(expr)
    local chunk, err = loadstring("return " .. expr)
    if chunk then
        local ok, result = pcall(chunk)
        if ok then return result end
        error(result)
    else
        error(err)
    end
end

print("eval('2+3'):", eval("2+3"))        -- 5
print("eval('math.max(1,5,3)'):", eval("math.max(1,5,3)"))  -- 5
```

## 4. 元表反射

### 代码示例

```lua
-- __index 拦截（代理对象）
local function create_proxy(original)
    return setmetatable({}, {
        __index = function(self, key)
            print("accessing:", key)
            return original[key]
        end,
        __newindex = function(self, key, value)
            print("setting:", key, "=", value)
            original[key] = value
        end,
    })
end

local data = {a = 1, b = 2}
local proxy = create_proxy(data)

print("proxy.a:", proxy.a)  -- accessing: a  →  1
proxy.c = 3                 -- setting: c  =  3
```

## 5. 环境 _ENV

### 概念讲解

`_ENV`：当前环境，搜索全局变量的表
`_G`：全局表（初始的 _ENV）

### 代码示例

```lua
local function create_sandbox()
    local env = {
        print = print,
        math = math,
        table = table,
    }
    setmetatable(env, {__index = _G})
    return env
end

local sandbox = create_sandbox()
local code = [[print(math.random(100))]]

local chunk = loadstring(code)
if chunk then
    setfenv(chunk, sandbox)
    chunk()
end
```
