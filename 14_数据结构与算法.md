## 1. 链表

### 代码示例

```lua
-- 单向链表
local function create_list(...)
    local head = nil
    for i = select("#", ...), 1, -1 do
        head = {value = select(i, ...), next = head}
    end
    return head
end

-- 反转链表
local function reverse(head)
    local prev = nil
    local current = head
    while current do
        local next = current.next
        current.next = prev
        prev = current
        current = next
    end
    return prev
end

local head = create_list(1, 2, 3, 4, 5)
head = reverse(head)
-- 遍历
local result = {}
local current = head
while current do
    table.insert(result, current.value)
    current = current.next
end
print(table.concat(result, ", "))  -- 5, 4, 3, 2, 1
```

## 2. 栈和队列

### 代码示例

```lua
-- 栈（LIFO）
local function create_stack()
    return {top = 0, data = {}}
end
local function push(s, v) s.top = s.top + 1; s.data[s.top] = v end
local function pop(s) if s.top <= 0 then return nil end; local v = s.data[s.top]; s.top = s.top - 1; return v end

-- 队列（FIFO，注意 table.remove(1) 是 O(n)）
local function create_queue()
    return {head = 0, tail = 0, data = {}}
end
local function enqueue(q, v) q.data[q.tail] = v; q.tail = q.tail + 1 end
local function dequeue(q) if q.head >= q.tail then return nil end; local v = q.data[q.head + 1]; q.head = q.head + 1; return v end
```

## 3. 二叉搜索树（BST）

### 代码示例

```lua
local function insert(root, v)
    if not root then return {value = v, left = nil, right = nil} end
    if v < root.value then root.left = insert(root.left, v)
    elseif v > root.value then root.right = insert(root.right, v) end
    return root
end

local function search(root, v)
    if not root then return nil end
    if v == root.value then return root
    elseif v < root.value then return search(root.left, v)
    else return search(root.right, v) end
end

local function inorder(root)
    if not root then return {} end
    local r = {}
    local function trav(n)
        if n.left then trav(n.left) end
        table.insert(r, n.value)
        if n.right then trav(n.right) end
    end
    trav(root)
    return r
end

local root = nil
for _, v in ipairs({5, 3, 7, 1, 4, 6, 8}) do root = insert(root, v) end
print(table.concat(inorder(root), ", "))  -- 1, 3, 4, 5, 6, 7, 8
print("search 4:", search(root, 4) ~= nil)  -- true
```

## 4. 排序算法

### 代码示例

```lua
-- 快速排序
local function quick_sort(arr)
    if #arr <= 1 then return arr end
    local pivot = arr[1]
    local lower, higher = {}, {}
    for i = 2, #arr do
        if arr[i] < pivot then table.insert(lower, arr[i])
        else table.insert(higher, arr[i]) end
    end
    local sorted = quick_sort(lower)
    table.insert(sorted, pivot)
    for _, v in ipairs(quick_sort(higher)) do table.insert(sorted, v) end
    return sorted
end

print(table.concat(quick_sort({5, 2, 8, 1, 9, 3}), ", "))  -- 1, 2, 3, 5, 8, 9
```

## 5. 二分查找

### 代码示例

```lua
local function binary_search(arr, target)
    local left, right = 1, #arr
    while left <= right do
        local mid = math.floor((left + right) / 2)
        if arr[mid] == target then return mid
        elseif arr[mid] < target then left = mid + 1
        else right = mid - 1 end
    end
    return nil
end

local sorted = {1, 3, 5, 7, 9, 11}
print("find 7:", binary_search(sorted, 7))  -- 4
print("find 8:", binary_search(sorted, 8))  -- nil
```

## 6. 图算法（BFS/DFS）

### 代码示例

```lua
-- 邻接表
local graph = {
    [1] = {2, 3}, [2] = {4, 5}, [3] = {6},
    [4] = {}, [5] = {6}, [6] = {},
}

-- DFS
local function dfs(g, start, visited)
    visited = visited or {}
    visited[start] = true
    print("visit:", start)
    for _, n in ipairs(g[start] or {}) do
        if not visited[n] then dfs(g, n, visited) end
    end
end

-- BFS
local function bfs(g, start)
    local visited = {[start] = true}
    local queue = {start}
    while #queue > 0 do
        local node = table.remove(queue, 1)
        print("visit:", node)
        for _, n in ipairs(g[node] or {}) do
            if not visited[n] then
                visited[n] = true
                table.insert(queue, n)
            end
        end
    end
end

print("DFS from 1:")
dfs(graph, 1)
print("BFS from 1:")
bfs(graph, 1)
```

|--------------|---------|
| 链表 | table + next 指针 |
| 栈 | table + top 指针 |
