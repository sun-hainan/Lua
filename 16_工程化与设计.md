## 1. 设计模式

### 代码示例

```lua
-- 模块模式
local function create_module()
    local _private_var = 0
    return {
        get = function() return _private_var end,
        set = function(v) _private_var = v end,
    }
end

-- 单例模式
local function Singleton()
    local instance = nil
    return function()
        if not instance then
            instance = {value = "singleton"}
        end
        return instance
    end
end

-- 工厂模式
local function create_shape(t)
    if t == "circle" then
        return {area = function(self) return math.pi * self.radius^2 end, radius = 1}
    elseif t == "rect" then
        return {area = function(self) return self.w * self.h end, w = 2, h = 3}
    end
end

-- 观察者模式
local function create_observer()
    local observers = {}
    return {
        subscribe = function(fn) table.insert(observers, fn) end,
        notify = function(data)
            for _, fn in ipairs(observers) do fn(data) end
        end,
    }
end
```

## 2. 代码优化技巧

### 代码示例

```lua
-- 局部变量缓存（避免重复查找）
local function fast_lookup()
    local sin, cos, tan = math.sin, math.cos, math.tan
    for i = 1, 1000 do
        local v = sin(cos(tan(i)))
    end
end

-- table.concat 替代字符串拼接
local function fast_concat()
    local parts = {}
    for i = 1, 100 do
        parts[i] = "item" .. i
    end
    return table.concat(parts, ",")
end
```

## 3. 单元测试

### 代码示例

```lua
local function assert(condition, msg)
    if not condition then error("Assertion failed: " .. (msg or "")) end
end

local function assert_equal(expected, actual, msg)
    if expected ~= actual then
        error(string.format("Expected %s but got %s%s",
            tostring(expected), tostring(actual), msg and ": " .. msg or ""))
    end
end

local function test(name, fn)
    local ok, err = pcall(fn)
    if ok then print("PASS:", name)
    else print("FAIL:", name, err) end
end

test("add(2,3) = 5", function()
    local function add(a, b) return a + b end
    assert_equal(5, add(2, 3))
end)
```

## 4. 错误处理最佳实践

### 代码示例

```lua
local function safe_divide(a, b)
    local ok, result = pcall(function()
        if b == 0 then error("division by zero", 2) end
        return a / b
    end)
    if ok then return result, nil
    else return nil, result end
end

-- 返回 (nil, error) 模式
local function parse_number(s)
    local n = tonumber(s)
    if n then return n, nil
    else return nil, "invalid number" end
end
```

## 5. 模块系统

### 代码示例

## 6. Git 和 CI/CD

### Git 基础

```bash
git init
git add .
git commit -m 'feat: add new feature'
git push origin main
git checkout -b feature/new-feature
```

### GitHub Actions

```yaml
# .github/workflows/lua.yml
name: Lua CI
on: [push, pull_request]
jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Setup Lua
        uses: leafo/gh-actions-lua@v2
      - name: Run Tests
        run: busted
      - name: Lint
        run: luacheck .
```
