## 1. 观察者模式

### 生活类比

> 观察者模式就像"订阅报纸"——你（观察者）去报社（主题）登记订阅，报社有新消息就通知你。不想要了就去退订。

### 代码示例

```lua
-- 主题
local EventSubject = {}
function EventSubject:new()
    return setmetatable({observers = {}}, {__index = self})
end
function EventSubject:subscribe(obs)
    table.insert(self.observers, obs)
end
function EventSubject:unsubscribe(i)
    table.remove(self.observers, i)
end
function EventSubject:notify(event)
    for _, obs in ipairs(self.observers) do
        obs:update(event)
    end
end

-- 具体观察者
local Logger = {update = function(self, e)
    print("[LOG]", e.message, "at", e.timestamp) end}

local EmailNotifier = {update = function(self, e)
    print("[EMAIL] Would send:", e.message) end}

local subject = EventSubject:new()
subject:subscribe(Logger)
subject:subscribe(EmailNotifier)
subject:notify({message = "User logged in", timestamp = os.date("%H:%M:%S")})
```

## 2. EventEmitter

### 代码示例

```lua
local EventEmitter = {}
function EventEmitter:new()
    return setmetatable({listeners = {}}, {__index = self})
end
function EventEmitter:on(event, handler)
    if not self.listeners[event] then self.listeners[event] = {} end
    table.insert(self.listeners[event], handler)
end
function EventEmitter:once(event, handler)
    local wrapped = function(...) handler(...); self:remove_listener(event, wrapped) end
    self:on(event, wrapped)
end
function EventEmitter:emit(event, ...)
    if self.listeners[event] then
        for _, h in ipairs(self.listeners[event]) do h(...) end
    end
end
function EventEmitter:remove_listener(event, handler)
    if self.listeners[event] then
        for i, h in ipairs(self.listeners[event]) do
            if h == handler then table.remove(self.listeners[event], i) break end
        end
    end
end

local emitter = EventEmitter:new()
emitter:on("data", function(d) print("handler1 got:", d) end)
emitter:on("data", function(d) print("handler2 got:", d) end)
emitter:emit("data", "hello world")

emitter:once("single", function() print("This fires only once") end)
emitter:emit("single")
emitter:emit("single")  -- 不触发
```

## 3. 消息队列

### 代码示例

```lua
-- 简单队列
local function create_queue() return {data = {}, head = 1} end
local function enqueue(q, item) table.insert(q.data, item) end
local function dequeue(q)
    if q.head > #q.data then return nil end
    return table.remove(q.data, q.head)
end

local q = create_queue()
enqueue(q, "task1")
enqueue(q, "task2")
enqueue(q, "task3")
print(dequeue(q))  -- task1
print(dequeue(q))  -- task2

-- 优先级队列
local function pq_enqueue(q, priority, item)
    table.insert(q, {priority = priority, item = item})
    for i = #q, 2, -1 do
        if q[i].priority < q[i-1].priority then q[i], q[i-1] = q[i-1], q[i] end
    end
end
```

## 4. 中间件模式

### 代码示例

```lua
local function create_pipeline()
    local middlewares = {}
    return {
        use = function(self, mw) table.insert(middlewares, mw) end,
        execute = function(self, context, final)
            local function run(i)
                if i > #middlewares then return final(context) end
                middlewares[i](context, function() run(i + 1) end)
            end
            return run(1)
        end,
    }
end

local logger = function(ctx, next) print("LOG: before"); next(); print("LOG: after") end
local auth = function(ctx, next)
    if ctx.authenticated then print("AUTH: ok"); next()
    else print("AUTH: denied") end
end
local handler = function(ctx) print("HANDLER: processing", ctx.path) end

local p = create_pipeline()
p:use(logger)
p:use(auth)
p:execute({path = "/api", authenticated = true}, handler)
```

## 5. Actor 模型

### 代码示例

```lua
local function create_actor(behavior)
    local mailbox = {}
    return {
        send = function(self, msg) table.insert(mailbox, msg) end,
        run = function(self)
            while #mailbox > 0 do
                behavior(table.remove(mailbox, 1))
            end
        end,
    }
end

local counter = create_actor(function(msg)
    counter.state = (counter.state or 0) + 1
    print("count:", counter.state)
end)

counter:send("increment")
counter:send("increment")
counter:send("increment")
counter:run()
```

## 6. 响应式流（协程实现）

### 代码示例

```lua
local function range(from, to)
    return coroutine.create(function()
        for i = from, to do coroutine.yield(i) end
    end)
end

local function filter(co, pred)
    return coroutine.create(function()
        while true do
            local ok, val = coroutine.resume(co)
            if not ok then break end
            if pred(val) then coroutine.yield(val) end
        end
    end)
end

local function collect(co)
    local r = {}
    while true do
        local ok, val = coroutine.resume(co)
        if not ok then break end
        table.insert(r, val)
    end
    return r
end

local nums = range(1, 10)
local evens = filter(nums, function(n) return n % 2 == 0 end)
local doubled = filter(evens, function(n) coroutine.yield(n * 2) end)
print(table.concat(collect(doubled), ", "))  -- 4, 8, 12, 16, 20
```
